230  // partitions constraint vectors into contact and limit constraints
231  void partition_constraints()
232  {
233  const unsigned UINF = std::numeric_limits<unsigned>::max();
234  contact_constraints.clear();
235  limit_constraints.clear();
236 
237  BOOST_FOREACH(UnilateralConstraint* e, constraints)
238  {
239  if (e->constraint_type == UnilateralConstraint::eContact)
240  contact_constraints.push_back(e);
241  else
242  {
243  assert(e->constraint_type == UnilateralConstraint::eLimit);
244  limit_constraints.push_back(e);
245  }
246  }
247 
248  // now, sort the contact constraints such that constraints that use a true friction
249  // cone are at the end
250  if(!contact_constraints.empty()){
251  for (unsigned i=0, j=contact_constraints.size()-1; i< j; )
252  {
253  if (contact_constraints[i]->contact_NK == UINF)
254  {
255  std::swap(contact_constraints[i], contact_constraints[j]);
256  j--;
257  }
258  else
259  i++;
260  }
261  }
262  }

264  // sets stacked vector from cn, cs, etc.
265  Ravelin::VectorNd& to_stacked(Ravelin::VectorNd& z)
266  {
267  z.set_sub_vec(CN_IDX, cn);
268  z.set_sub_vec(CS_IDX, cs.segment(0, N_LIN_CONE));
269  z.set_sub_vec(CT_IDX, ct.segment(0, N_LIN_CONE));
270  z.set_sub_vec(NCS_IDX, cs.segment(N_LIN_CONE, cs.size()));
271  z.set_sub_vec(NCT_IDX, ct.segment(N_LIN_CONE, ct.size()));
272  for (unsigned i=CS_IDX, j=NCS_IDX; i< CT_IDX; i++, j++)
273  {
274  if (z[i] < 0.0)
275  {
276  z[j] = -z[i];
277  z[i] = 0.0;
278  }
279  else
280  z[j] = 0.0;
281  }
282  for (unsigned i=CT_IDX, j=NCT_IDX; i< NCS_IDX; i++, j++)
283  {
284  if (z[i] < 0.0)
285  {
286  z[j] = -z[i];
287  z[i] = 0.0;
288  }
289  else
290  z[j] = 0.0;
291  }
292  z.set_sub_vec(L_IDX, l);
293  return z;
294  }
